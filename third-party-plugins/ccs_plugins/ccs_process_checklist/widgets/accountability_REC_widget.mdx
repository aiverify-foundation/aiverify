export const ib_cid = "accountability";

  const responses = data02.map(item => item.name);
  const yesCount = data02.find(item => item.name === 'Yes')?.value || 0;
  const noCount = data02.find(item => item.name === 'No')?.value || 0;
  const naCount = data02.find(item => item.name === 'Not Applicable')?.value || 0; 

  let recommendation;
  if (noCount >= 1) {
    //Failed Test
    recommendation = `The market would trend towards positive outcomes such as the bolstering of innovation and competitive dynamics, if there were mechanisms to determine the proper allocation of accountability and responsibility. Proper allocation of accountability will also incentivise firms to improve consumer outcomes, and consumers may more easily seek redress when things go wrong, leading to consumer confidence, trust and adoption.
    The business should take proactive responsibility for what they control in the value chain and take positive action necessary to ensure consumers are adequately protected and ensure the development of a competitive market that gains the trust and confidence of consumers and businesses. The business should also document all follow-up actions taken when their AI systems are harmful to the market and repeat the self-assessment periodically to ensure that any identified competition or consumer protection issues have been resolved.`;
  } else {
    //Passed Test
    recommendation = `The market would trend towards positive outcomes, such as the bolstering of innovation and competitive dynamics, if there were mechanisms to determine the proper allocation of accountability and responsibility. Proper allocation of accountability will also incentivise firms to improve consumer outcomes, and consumers may more easily seek redress when things go wrong, leading to consumer confidence, trust and adoption.
    The business has ensured that accountability for any issues arising from the AI model are allocated adequately and that consumers can easily seek redress issues for problems with the business’s AI products.`;
  }

  return (
    //Attempt to new line
    <div className="recommendation">
      <h3 className="c-primary">Our Recommendation:</h3>
      <div style={{paddingTop: "5px"}}>
        {recommendation}
      </div>
    </div>
  );
};

export const MissingCriteria = ({ data }) => {
    const questionsDict = {
        "7_1_1": "7.1.1 Ensure that the business has the accountability and incentive to improve consumers’ experience.",
        "7_1_2": "7.1.2 Ensure that clear information about routes to redress are provided to consumers for issues that can occur."
    };

    const noResponses = data
        .filter(response => response.response === 'No')
        .map(response => response.id);

    const missingCriteriaQuestions = noResponses.map(id => questionsDict[id] || "Question not found");

    return (
        <div style={{ paddingLeft: "10px", paddingRight: "10px", paddingTop: "15px" }}>
            <h3 className="c-primary">Business did not implement the following testable criteria fully:</h3>
            <ul style={{ paddingLeft: "13px", paddingRight: "0px", marginTop: "5px" }}>
                {missingCriteriaQuestions.map((question, index) => (
                    <li key={index} style={{ paddingBottom: "9px" }}>
                        {question}
                    </li>
                ))}
            </ul>
        </div>
    );
}

        const noCount = responses.filter(response => response === 'No').length;
        const naCount = responses.filter(response => response === 'NA').length;

        setData02([
            { name: "Not Applicable", value: naCount }, // Blue

// for grid display - display: "grid", gridTemplateColumns: "1fr 1fr", gap: "0px", padding: "0px"

    return (

The market would trend towards positive outcomes, such as the bolstering of innovation and competitive dynamics, if there were mechanisms to determine the proper allocation of accountability and responsibility. Proper allocation of accountability will also incentivise firms to improve consumer outcomes, and consumers may more easily seek redress when things go wrong, leading to consumer confidence, trust and adoption.

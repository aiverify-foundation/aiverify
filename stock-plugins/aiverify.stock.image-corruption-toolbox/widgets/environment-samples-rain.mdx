import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_environment_corruptions";
import {
  fog_images,
  snow_images,
  rain_images,
} from "./environment-corruption-images.js";

export const RainSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const rainUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Rain'))
    : rain_images;

  return (
    <DisplayImages
      id={props.id}
      images={rainUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={2}
      corruption_name="Environment: Rain"
    />
  );
}

<RainSamples {...props} />

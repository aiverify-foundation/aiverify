import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_environment_corruptions";
import {
  fog_images,
  snow_images,
  rain_images,
} from "./environment-corruption-images.js";

export const FogSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const fogUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Fog'))
    : fog_images;

  return (
    <DisplayImages
      id={props.id}
      images={fogUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={0}
      corruption_name="Environment: Fog"
    />
  );
}

<FogSamples {...props} />

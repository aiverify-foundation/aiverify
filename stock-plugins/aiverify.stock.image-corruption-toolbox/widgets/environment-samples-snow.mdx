import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_environment_corruptions";
import {
  fog_images,
  snow_images,
  rain_images,
} from "./environment-corruption-images.js";

export const SnowSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const snowUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Snow'))
    : snow_images;

  return (
    <DisplayImages
      id={props.id}
      images={snowUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={1}
      corruption_name="Environment: Snow"
    />
  );
}

<SnowSamples {...props} />

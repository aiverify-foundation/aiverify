import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_digital_corruptions";
import {
  brightness_down_images,
  brightness_up_images,
  contrast_down_images,
  contrast_up_images,
  saturate_images,
  random_tilt_images,
  compression_images,
} from "./digital-corruption-images.js";

export const TiltSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const tiltUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Tilt'))
    : random_tilt_images;

  return (
    <DisplayImages
      id={props.id}
      images={tiltUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={5}
      corruption_name="Digital: Tilt"
    />
  );
}

<TiltSamples {...props} />

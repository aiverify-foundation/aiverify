  gaussian_general_images,

export const SNPSamples = (props) => {

  const snpUrls = artifactUrls.length > 0

  return (
    <DisplayImages

      images={snpUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={2}
      corruption_name="General: Salt and Pepper Noise"
    />
  );
}

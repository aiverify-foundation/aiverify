import { CorruptionBarChart } from './corruption-bar-chart'
export const algo_cid = "aiverify_digital_corruptions"

export const DigitalBarChart = ({ props }) => {
  const resultsData = props.getResults ? props.getResults(algo_cid).results : null;
  
  if (!resultsData || resultsData.length === 0) {
    return <div>No results data available</div>;
  }

  return (
    <CorruptionBarChart
      data={resultsData}
      width={props.width || "100%"}
    />
  );
};

<DigitalBarChart props={props} />

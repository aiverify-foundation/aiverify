import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_blur_corruptions";
import {
  gaussian_blur_images,
  glass_blur_images,
  defocus_blur_images,
  hmotion_blur_images,
  vmotion_blur_images,
  zoom_blur_images,
} from "./blur-corruption-images.js";

export const DefocusSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const defocusUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Defocus_Blur'))
    : defocus_blur_images;

  return (
    <DisplayImages
      id={props.id}
      images={defocusUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={1}
      corruption_name="Blur: Defocus Blur"
    />
  );
}

<DefocusSamples {...props} />

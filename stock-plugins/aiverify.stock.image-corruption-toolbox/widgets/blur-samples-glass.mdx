import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_blur_corruptions";
import {
  gaussian_blur_images,
  glass_blur_images,
  defocus_blur_images,
  hmotion_blur_images,
  vmotion_blur_images,
  zoom_blur_images,
} from "./blur-corruption-images.js";

export const GlassSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const glassUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Glass_Blur'))
    : glass_blur_images;

  return (
    <DisplayImages
      id={props.id}
      images={glassUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={1}
      corruption_name="Blur: Glass Blur"
    />
  );
}

<GlassSamples {...props} />

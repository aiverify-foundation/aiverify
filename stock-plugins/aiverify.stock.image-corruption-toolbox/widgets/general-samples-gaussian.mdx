import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_general_corruptions";
import {
  gaussian_general_images,
  poisson_general_images,
  snp_general_images,
} from "./general-corruption-images.js";

export const GaussianSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const gaussianUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Gaussian_Noise'))
    : gaussian_general_images;

  return (
    <DisplayImages
      id={props.id}
      images={gaussianUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={0}
      corruption_name="General: Gaussian Noise"
    />
  );
}

<GaussianSamples {...props} />

import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_blur_corruptions";
import {
  gaussian_blur_images,
  glass_blur_images,
  defocus_blur_images,
  hmotion_blur_images,
  vmotion_blur_images,
  zoom_blur_images,
} from "./blur-corruption-images.js";

export const ZoomSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const zoomUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Zoom_Blur'))
    : zoom_blur_images;

  return (
    <DisplayImages
      id={props.id}
      images={zoomUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={5}
      corruption_name="Blur: Zoom Blur"
    />
  );
}

<ZoomSamples {...props} />

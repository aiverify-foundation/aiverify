import React, { useEffect, useState } from "react";
import { DisplayImages } from "./display-samples.mdx";

export const algo_cid = "aiverify_digital_corruptions";
import {
  brightness_down_images,
  brightness_up_images,
  contrast_down_images,
  contrast_up_images,
  saturate_images,
  random_tilt_images,
  compression_images,
} from "./digital-corruption-images.js";

export const CompressionSamples = (props) => {
  const artifactUrls = props.getArtifacts(algo_cid);
  const compressionUrls = artifactUrls.length > 0 
    ? artifactUrls.filter(url => url.includes('Compression'))
    : compression_images;

  return (
    <DisplayImages
      id={props.id}
      images={compressionUrls}
      data={props.getResults(algo_cid).results}
      corruption_index={6}
      corruption_name="Digital: Compression"
    />
  );
}

<CompressionSamples {...props} />
